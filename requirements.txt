# SRS-SDK-002 — trading-calendar-aware scheduling.
# exchange_calendars supplies NYSE/NASDAQ holiday and early-close data;

# so the SRS-SDK-001 parity AST scan does not see the vendor imports.
exchange_calendars>=4.5,<5
croniter>=2.0,<3

# SRS-SDK-006 — built-in technical indicators wrap pandas-ta and TA-Lib.
# SyRS SYS-35 + AC-6 + StRS C-9 require the indicator library to wrap
# these two reference implementations; "custom reimplementations of
# indicators available in these libraries are prohibited" (AC-6).
#
# Version pins are deliberately tight so the runtime never resolves to an
# incompatible combination:
#   - pandas-ta 0.4.71b0 is NumPy-2 compatible (0.3.14b is NOT — it pins
#     numpy<2 transitively, blocking the import chain).
#   - pandas-ta pulls in numba transitively, which does NOT yet support
#     Python 3.14. python/pyproject.toml pins requires-python = ">=3.12,<3.14"
#     so users on a supported interpreter always see a working import chain.
#   - TA-Lib python wrapper 0.6.x is built against TA-Lib C library 0.6.x;
#     mixing wrapper 0.6.x with the older 0.4.0 C library leaves the
#     ctypes lookups dangling at import time. The Dockerfile +
#     ci.yml install TA-Lib C 0.6.4 to match.
# `brew install ta-lib` on macOS gets 0.6.x; matches the Docker layer.
numpy==2.2.6
pandas==2.3.3
pandas-ta==0.4.71b0
TA-Lib==0.6.8

# SRS-RES-002 — the Jupyter research environment renders plots.
# SyRS SYS-34b requires the research environment to have "plotting
# capabilities" alongside the unified data interface (SYS-27) and the
# indicator library (SYS-35); matplotlib is the canonical Python plotting
# backend (it is also what pandas' DataFrame.plot() uses). The
# atp_research.plotting helper renders through matplotlib's object-oriented
# Figure API (no interactive/GUI backend required, so it works headless in
# CI and inside the operator-supplied JupyterLab image alike).
matplotlib==3.11.0
